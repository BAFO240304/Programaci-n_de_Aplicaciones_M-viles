bool esPrimo(int numero) {
if (numero <= 1) return false;

Implementa un programa que genere la secuencia de Fibonacci hasta un n√∫mero n de t√©rminos ingresado por el usuario. Utiliza un bucle while o for para ir generando los n√∫meros de la secuencia.

void main() {
  print('Ingresa el n√∫mero de t√©rminos de la secuencia de Fibonacci:');
  int n = int.parse(stdin.readLineSync()!);
  if (n <= 0) {
    print('Por favor, ingresa un n√∫mero positivo.');
    return;
  }
  BigInt a = BigInt.zero;
  BigInt b = BigInt.one; 
  print('La secuencia de Fibonacci hasta $n t√©rminos es:');
  for (int i = 0; i < n; i++) {
    if (i == 0) {
      print(a);
      continue;
    }
    if (i == 1) {
      print(b);
      continue;
    }
    BigInt siguiente = a + b;
    print(siguiente);
    a = b;
    b = siguiente;
  }
}

import 'dart:io';
void main() {
  print('Ingresa un n√∫mero para calcular su factorial:');
  int n = int.parse(stdin.readLineSync()!);
  if (n < 0) {
    print('El factorial no est√° definido para n√∫meros negativos.');
    return;
  }
  BigInt factorial = BigInt.one;

  for (int i = 2; i <= n; i++) {

  print('El factorial de $n es: $factorial');
}

Ejercicio 7:

Crea un programa que imprima una matriz cuadrada de tama√±o n x n en forma de espiral. Utiliza bucles anidados para recorrer las posiciones de la matriz en el orden adecuado.


Codigo en Gif Hub üìã 

import 'dart:io';
void main() {
  print('Ingresa el tama√±o de la matriz (n):');
  int n = int.parse(stdin.readLineSync()!);
  List> matriz = List.generate(n, (i) => List.filled(n, 0));
  int num = 1;
  int inicioFila = 0, finFila = n - 1;
  int inicioCol = 0, finCol = n - 1;
  while (num <= n * n) {
    for (int i = inicioCol; i <= finCol; i++) {
      matriz[inicioFila][i] = num++;
    }
    inicioFila++;
    for (int i = inicioFila; i <= finFila; i++) {
      matriz[i][finCol] = num++;
    }
    finCol--;
    if (inicioFila <= finFila) {
      for (int i = finCol; i >= inicioCol; i--) {
        matriz[finFila][i] = num++;
      }
      finFila--;
    }
    if (inicioCol <= finCol) {
      for (int i = finFila; i >= inicioFila; i--) {
        matriz[i][inicioCol] = num++;
      }
      inicioCol++;
    }
  }
  print('Matriz en espiral:');
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      stdout.write('${matriz[i][j]} ');
    }
    print('');
  }
}
